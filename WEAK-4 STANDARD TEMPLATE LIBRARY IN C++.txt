    // front and rear
    cout << "Front element in q-" << q.front() << endl; // 20
    cout << "REAR element in q--" << q.back() << endl;  // 40

    {
        cout << "q is empty";
    }
    else
    {
        cout << "q is not empty";
    }

    // swapping

{

    priority_queue<int> pq;
    // max-heap-> max val->highest pq

    pq.push(10); // 10
    pq.push(23); // 23,10
    pq.push(50); // 50,23,10
    pq.push(34); // 50,34,23,10

    // top ele---highest pq
    cout << "top ele-" << pq.top() << endl; // 50

    // pop ele-highest pq remove pop
    pq.pop();
    cout << "top ele-" << pq.top() << endl; // 34

    // size
    cout << "size of pq-" << pq.size() << endl; // 3

    // empty

    {
        cout << "pq is mpty" << endl;
    }
    else
    {
        cout << "pq is not empty" << endl;
